### 思路
对于异或 + 查询类问题，基本使用字典树，先套板子。

刚开始没理解清题意，直接构建了树，啪地提交就wa了。

再仔细一看，需要算出每个queries中不大于mi的值的最大异或值，所以在对queries[i]进行查询时，构建的字典树中的最大值不能超过mi。

现在就十分明了，对nums排序、对queiries先记录位置，因为排序之后位置会乱，，然后对queiries进行二维的排序，用到java的自定义排序。
可参考[java中的自定义比较器](https://juejin.cn/post/6965104809011576846)

接下来就是边插入边查询了。。。
### 代码
```java
class TrieNode{
    int count; //表示以该处节点构成的串的个数
    int preCount; //表示以该处节点构成的前缀的字串的个数
    TrieNode[] children;

    TrieNode() {

        children = new TrieNode[2];
        count = 0;
        preCount = 0;
    }
}
class Trie{
    TrieNode root;

    public Trie() {
        root = new TrieNode();
    }

    public void insert(int val) {

        TrieNode node = root;

        for (int i = 29; i >= 0; i--) {
            int bit = (val >> i) & 1;
            if (node.children[bit] == null) {
                node.children[bit] = new TrieNode();
            }
            node = node.children[bit];
            node.preCount++;
        }
        node.count++;
    }
}
class Solution {
    
    static int L = 30;
    public String int2bit(int x){
        StringBuffer bits = new StringBuffer();
        for(int i = L; i >= 0; i--){
            if((x & (1 << i)) != 0){
                bits.append(1);
            }
            else {
                bits.append(0);
            }
        }
        return bits.toString();
    }

    public int[] maximizeXor(int[] nums, int[][] qs) {
        Trie trie = new Trie();
        Arrays.sort(nums);
        int[][] queries = new int[qs.length][3];
        for(int i = 0; i < qs.length; i++){
            queries[i][0] = qs[i][0];
            queries[i][1] = qs[i][1];
            queries[i][2] = i;
        }
        Arrays.sort(queries, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                if(o1[1] == o2[1]){
                    return o1[0] - o2[0];
                }
                return o1[1] - o2[1];
            }
        });
        int[] ans = new int[queries.length];
        int pos = 0;
        int idx = 0;
        for(int[] querie:queries){
            int num = querie[0], curMax = querie[1];
            while(idx < nums.length && nums[idx] <= curMax){
                trie.insert(nums[idx]);
                idx++;
            }
            if(idx == 0){
                ans[querie[2]] = -1;
                continue;
            }
            TrieNode cur = trie.root;
            int tmp = 0;
            for(int i = 29 ; i >= 0; i--){
                int bit = (num >> i) & 1;
                if(cur.children[1 - bit] != null){
                    tmp |= 1 << i;
                    bit ^= 1;
                }
                cur = cur.children[bit];
            }
            ans[querie[2]] = tmp;
        }
        return ans;
    }
}
```
